## 🎯 كيفية إضافة ميزات جديدة:

### مثال عملي: إضافة صفحة إدارة الطلاب

#### الخطوة 1: إنشاء الصفحة
```typescript
// frontend/src/pages/StudentsManagement.tsx

import { useState, useEffect } from 'react'

export default function StudentsManagement() {
  const [students, setStudents] = useState([])
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    fetchStudents()
  }, [])

  const fetchStudents = async () => {
    try {
      const response = await fetch('http://localhost:5000/api/v1/students')
      const data = await response.json()
      setStudents(data.students)
    } catch (error) {
      console.error('Error:', error)
    } finally {
      setLoading(false)
    }
  }

  return (
    <div className="dashboard">
      <div className="dashboard-header">
        <h1>إدارة الطلاب</h1>
      </div>

      <div className="dashboard-content">
        <table>
          <thead>
            <tr>
              <th>الاسم</th>
              <th>الجنسية</th>
              <th>الحلقة</th>
              <th>الحالة</th>
            </tr>
          </thead>
          <tbody>
            {students.map(student => (
              <tr key={student.id}>
                <td>{student.profile?.firstName} {student.profile?.lastName}</td>
                <td>{student.profile?.nationality}</td>
                <td>{student.className}</td>
                <td>{student.isActive ? 'نشط' : 'متوقف'}</td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  )
}
```
